Example 1

Input:

# duplicates are not allowed
# 1,1.0,True are treated same
# 0,0.0, False are treated same
d = {1: 'A', 1.0: "F", True: 'B', 0: 'Z', 0.0: 'FZ', False: 'BZ'}
print(d)

Output:

{1: 'B', 0: 'BZ'}

Example 2

Input:

d = {"1": "A", 1.0: "BZ", True: "B"}
print(d)

Output:

{'1': 'A', 1.0: 'B'}

Example 3

Input:

a = "Venky"
d = {a: 12}
print(d)
a = "RBR"
print(d)

Output:

{'Venky': 12}
{'Venky': 12}

# dict method

Example 8

Input:

# copy a dictionary
d1 = {'A': 12, 'B': 20, 'C': 30}
d2 = d1 # dictionary is mutable
print(id(d2))
print(id(d1))

Output:

135765182677120
135765182677120

Example 9

Input:

d1['A'] = 100
print(d2)
print(d1)

Output:

{'A': 12, 'B': 20, 'C': 30}
{'A': 100, 'B': 20, 'C': 30}

Example 10

Input:

d3 = d1.copy()
d1['A'] = 1000
