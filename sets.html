Example 1

Input:

set_1 = {1,2,3,4,5}
print(set_1)
print(type(set_1))

Output:

{1, 2, 3, 4, 5}
class 'set'

Example 2

Input:

set_2 = set((1,2,3,4,5))
print(set_2)
print(type(set_2))

Output:

{1, 2, 3, 4, 5}
class 'set'

Example 3

Input:

# set is unordered, so indexing is not applicable
set_3 = {1,4,2,2.0,2.5,3.7,3,"Virat", "Rajini", "RBR"}
print(set_3)

Output:

{1, 2.5, 2, 3.7, 4, 3, 'RBR', 'Rajini', 'Virat'}

Example 4

Input:

# set will contain unique values only
set_3.add("Rohit")
print(set_3)

Output:

{1, 2.5, 2, 3.7, 4, 3, 'RBR', 'Rajini', 'Virat', 'Rohit'}

Example 5

Input:

# cannot use indexing
set_3[0]

Output:

set_3[0]
TypeError: 'set' object is not subscriptable

l[-1]

Example 7

Input:

set_b = {"1","2",1,2}
set_b.add(["A","B"])

Output:

set_b.add(["A","B"])
TypeError: unhashable type: 'list'

Example 8

Input:

set_b = {"1","2",1,2}
set_b.add(("A","B"))
print(set_b)

Output:

{1, 2, '2', '1', ('A', 'B')}

Example 9

Input:

set_c = {1,2,('A','B'),{1,2,3}}
print(set_c)

Output:

set_c = {1,2,('A','B'),{1,2,3}}
print(set_c)
