Example 1

Input:

# 1 , 1.0, True
# 0, 0.0, False
# above ones are treated as same in set
s = {1, True}
print(s)

Output:

{1}

Example 2

hash(1)
hash(True)

Output:

1
1

Example 3

hash((1,2,3,[12,3]))

TypeError: unhashable type: 'list'

Example 4

len({1,2,3,5,7,10,5})

Example 5

Input:

# accessing a set
s = {"Venky", "Rajini", "Kamal", "Allu Arjun", "Ramcharan"}
s[0]

Output:

TypeError: 'set' object is not subscriptable

Example 5

Input:

"Venky" in s

Output:

True

Example 7

Input:

s = {1,2,5.4,"abc", "1234",2.3,5}
print(s)

Output:

{1, 2, 2.3, '1234', 5, 5.4, 'abc'}

Example 8

Input:

# adding a element to set and updating a set
s = {10,20,30}
s.add(40)
print(s)

Output:

{40, 10, 20, 30}

Example 9

Input:

s.add(70,50,60)
print(s)
print(s)

Output:

TypeError: set.add() takes exactly one argument (3 given)

Example 10

Input:

s.add((70,50,60))
print(s)
s.add((10,20,30))
print(s)

Output:

{40, 10, 20, (70, 50, 60), 30}
{40, 10, (10, 20, 30), 20, (70, 50, 60), 30}
